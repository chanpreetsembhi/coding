How to Control Clean and organized all routes like `GET`, `POST`, `PUT`, `DELETE`.
- Create a controllers folder inner create a file `postController.js`.

```js
let posts = [
  { id: 1, title: "Post One" },
  { id: 2, title: "Post Two" },
  { id: 3, title: "Post Three" }
];

/* @dec   -  Get All Posts  */
/* @route -  GET /api/posts */
export const getPosts = (req, res) => {
  const limit = parseInt(req.query.limit);

  if (!isNaN(limit) && limit > 0) {
    return res.status(200).json(posts.slice(0, limit));
  }
  res.status(200).json(posts);
};

/* @dec   -  Get Single Post */
/* @route -  GET /api/posts/:id  */
export const getPost = (req, res, next) => {
  const id = parseInt(req.params.id);
  const post = posts.find((post) => post.id === id);

  if (!post) {
    const error = new Error(`A post with id of ${id} was not found`);
    error.status = 404;
    return next(error);
  }
  res.status(200).json(post);
};

/* @dec   -  Create A New Post */
/* @route -  POST /api/posts  */
export const createPost = (req, res, next) => {
  const newPost = {
    id: posts.length + 1,
    title: req.body.title,
  };

  if (!newPost.title) {
    const error = new Error(`Please include a title`);
    error.status = 400;
    return next(error);
  }
  posts.push(newPost);
  res.status(201).json(posts);
};

/* @dec   -  Update A New Post */
/* @route -  PUT /api/posts/:id  */
export const updatePost = (req, res, next) => {
  const id = parseInt(req.params.id);
  const post = posts.find((post) => post.id === id);

  if (!post) {
    const error = new Error(`A post with id of ${id} was not found`);
    error.status = 404;
    return next(error);
  }
  post.title = req.body.title;
  res.status(200).json(posts);
};

/* @dec   -  Delete A New Post */
/* @route -  DELETE /api/posts/:id  */
export const deletePost = (req, res, next) => {
  const id = parseInt(req.params.id);
  const post = posts.find((post) => post.id === id);
  
  if (!post) {
    const error = new Error(`A post with id of ${id} was not found`);
    error.status = 404;
    return next(error);
  }
  posts = posts.filter((post) => post.id !== id);
  res.status(200).json(posts);
};
```

- `post.js` File Structure.

```js
import express from "express";

import { createPost, deletePost, getPost, getPosts, updatePost } from "../controllers/postController.js";

const router = express.Router();


/* Get All Posts */
router.get("/", getPosts);

/* Get Single Posts */
router.get("/:id", getPost);

/* Create new post */
router.post("/", createPost);

/* Update Posts */
router.put("/:id", updatePost);

/* Delete Posts */
router.delete("/:id", deletePost); 

export default router;
```

import express from "express";

import blogs from "./routes/blogs.js";  

const app = express();
