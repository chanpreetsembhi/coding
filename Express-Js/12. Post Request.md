How to create a Post request with `router.post()` method.

**Note:** When you reload post data which you add later is gone because there is no database connect.

`posts.js` File Structure.

```js
import express from "express";

const router = express.Router();

let posts = [
  { id: 1, title: "Post One" },
  { id: 2, title: "Post Two" },
  { id: 3, title: "Post Three" },
];

/* Get All Posts */
router.get("/", (req, res) => {
  const limit = parseInt(req.query.limit);

  if (!isNaN(limit) && limit > 0) {
    return res.status(200).json(posts.slice(0, limit));
  }
  res.status(200).json(posts);
});

/* Create New Post */
router.post("/", (req, res) => {
  const newPost = {
    id: posts.length + 1,
    title: req.body.title,
  };

  if (!newPost.title) {
    return res.status(404).json({ Message: "Please include a title" });
  }
  posts.push(newPost);
  res.status(201).json(posts);
});

export default router;
```

`server.js` File Structure.
- parser a middleware.

```js
import express from "express";

import posts from "./routes/posts.js";

const port = process.env.PORT || 8000;

const app = express();

/* Body parser middleware */
app.use(express.json());
app.use(express.urlencoded({ extended: false }));

/* Routes */
app.use("/api/posts", posts);

app.listen(port, () => console.log(`Server is running on port ${port}`));
```
