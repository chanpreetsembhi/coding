#### Create a `package.json` file.

```
npm init

package name: default
version: (1.0.0)
description: Node Js
entry point: (index.js)
test command:
git repository:
keywords:
author: Your name
license: (ISC) MIT

{
  "name": "js",
  "version": "1.0.0",
  "description": "Node.js crash course code",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "author": "Chanpreet Singh",
  "license": "MIT"
}


Is this OK? (yes)
```
Hit Enter and your `package.json` file is ready.

==or skip question.==

```
npm init -y
```
