#### Alert

How to used it and style it. [official Docs](https://reactnative.dev/docs/alert)
- `index.tsx` File Structure.
- You can add buttons and style it.

```tsx
import { AppText } from "@/components/AppText";
import { Button } from "@/components/Button";
import { Alert, View } from "react-native";

export default function InderScreen() {

  const handleOpenAlert = () => {
    Alert.alert("Waring", "Are you sure you want to preceed?", [
      { text: "Cancel", style: "cancel" },
      {
        text: "Confirm",
        style: "destructive",
        onPress: () => {
          console.log("Let's go!");
        },
      },
    ]);
  };

  return (
    <View className="flex-1 justify-center p-4">
      <AppText center>Index Screen</AppText>
      <Button
        title="Open Alert"
        theme="secondary"
        onPress={() => handleOpenAlert()}
      />
    </View>
  );
}
```

#### React Native Modal

How to use Model. [Official Docs](https://reactnative.dev/docs/modal)
- `index.tsx` File Structure.

```tsx
import { AppText } from "@/components/AppText";
import { Button } from "@/components/Button";
import { useState } from "react"
import { Modal, View } from "react-native";

export default function InderScreen() {
  const router = useRouter();
  
  const [modalVisible, setModalVisible] = useState(false);

  return (
    <View className="flex-1 justify-center p-4">
      <AppText center>Index Screen</AppText>

      <Button
        title="Open Modal"
        theme="secondary"
        onPress={() => setModalVisible(true)}
      />
      
      <Modal
        visible={modalVisible}
        animationType="slide"
        onRequestClose={() => setModalVisible(false)}
      >
        <View className="flex-1 items-center justify-center">
          <View className="p-12 rounded-lg bg-white">
            <AppText center size="heading">
              A custom styled modal!
            </AppText>
            <Button
              title="Close"
              theme="secondary"
              onPress={() => setModalVisible(false)}
            />
          </View>
        </View>
      </Modal>
    </View>
  );
}
```

#### Expo Router Modal

Create (tabs) folder and move the all files inner folder. Create`_layout.tsx` and `modal` in root app directory.
- `_layout.tsx` File Struture.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
        <Stack.Screen name="modal" options={{ presentation: "modal" }} />
      </Stack>
    </React.Fragment>
  );
}
```

- `modal.tsx` File Structure
    - This open as sider It is a expo modal.

```tsx
import { AppText } from "@/components/AppText";
import { View } from "react-native";

export default function ModalScreen() {
  return (
    <View className="flex-1 items-center p-4">
      <AppText center>Modal Screen</AppText>
    </View>
  );
}
```

- This is button of modal. Use this in any screen.
```tsx
import { AppText } from "@/components/AppText";
import { Button } from "@/components/Button";
import { View } from "react-native";

export default function InderScreen() {

  return (

    <View className="flex-1 justify-center p-4">

      <Link href="/modal" push asChild>
        <Button title="Open Router Modal" theme="secondary" />
      </Link>
    </View>
  );
}
```

#### Multiple screens in one modal

How to add multiple screens in one modal
- Create a folder `modal-with-stack` and add `_layout.tsx`, `index.tsx`, `nested.tsx` and so on which you want.

Root `_layout.tsx` file structure. Add `model-with-stack` in it.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";
  
export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
        <Stack.Screen name="modal" options={{ presentation: "modal" }} />
        <Stack.Screen
          name="modal-with-stack"
          options={{ headerShown: false, presentation: "modal" }}
        />
      </Stack>
    </React.Fragment>
  );
}
```

- In `modal-with-stack` inner.
    - Add Modal Link Button in inner home page. 
    - In `_layout.tsx` use `Stack` only to show inner screens.
    - In `index.tsx` Structure
    
```tsx
import { AppText } from "@/components/AppText";
import { Button } from "@/components/Button";
import { Link } from "expo-router";
import { View } from "react-native";
  
export default function ModalWithStackScreen() {
  return (
    <View className="flex-1 items-center p-4">
      <AppText center>Modal With Stack Screen</AppText>
      <Link href="/modal-with-stack/nested" push asChild>
        <Button title="Push to /modal-with-stack/nested" theme="secondary" />
      </Link>
    </View>
  );
}
```
