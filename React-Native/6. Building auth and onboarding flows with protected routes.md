#### Protected routes in practice

App flow is **login** user see the **tabs layout** and **logout** user see only **Sign In** screen.
- To doing this we use wrap our `StackScreen` into `Stack.Protected`.

Root `_layout.tsx` file structure.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";

const isLoggedIn = true

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Protected guard={isLoggedIn}>
          <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
        </Stack.Protected>
        <Stack.Protected guard={!isLoggedIn}>
          <Stack.Screen name="sign-in" />
        </Stack.Protected>
      </Stack>
    </React.Fragment>
  );
}
```

#### Route resolution order

app folder.
- Create `create-account` page in **app** folder.
- If `shouldCreateAccount` value is false then user only show **Create Account** page.
- `_layout.tsx` File Structure.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";
  
const isLoggedIn = false;
const shouldCreateAccount = false;

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Protected guard={isLoggedIn}>
          <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
        </Stack.Protected>
        <Stack.Protected guard={!isLoggedIn && !shouldCreateAccount}>
          <Stack.Screen name="sign-in" />
        </Stack.Protected>
        <Stack.Protected guard={shouldCreateAccount}>
          <Stack.Screen name="create-account" />
        </Stack.Protected>
      </Stack>
    </React.Fragment>
  );
}
```

#### Persisting auth state

In this auth state we use [`Zustand`](https://zustand-demo.pmnd.rs/) and expo [`securestore`](https://docs.expo.dev/versions/latest/sdk/securestore/).

```bash
npm install zustand expo-secure-store
```
==If expo install not working==

Create `authStore.ts` file.

```tsx
import { getItem, setItem, deleteItemAsync } from "expo-secure-store";
import { create } from "zustand";
import { createJSONStorage, persist } from "zustand/middleware";

type UserState = {
  isLoggedIn: boolean;
  shouldCreateAccount: boolean;
  logIn: () => void;
  logOut: () => void;
};
  
export const useAuthStore = create(
  persist<UserState>(
    (set) => ({
      isLoggedIn: false,
      shouldCreateAccount: false,
      
      logIn: () => {
        set((state) => {
          return {
            ...state,
            isLoggedIn: true,
          };
        });
      },

      logOut: () => {
        set((state) => {
          return {
            ...state,
            isLoggedIn: false,
          };
        });
      },
    }),
    {
      name: "auth-store",
      storage: createJSONStorage(() => ({
        setItem,
        getItem,
        removeItem: deleteItemAsync,
      })),
    }
  )
);
```

Add new `hasCompletedOnboarding` state in `_layout.tsx` file,

import { Stack } from "expo-router";
