**The Auth Flow**

`Attempt to navigate to /`
`Logged In`
`Yes --> Open /`
`No --> Redirect to /login /`

#### Add the auth redirect

Authentication [docs](https://docs.expo.dev/router/advanced/authentication/)
- Move all the files in **(protected)** folder and add **layout** file in root.
- Add **Login** file in app root folder
- Add **Redirect** condition in **(protected)** folder inner layout.

```tsx
import { Redirect, Stack } from "expo-router";
import React from "react";

const isLoggedIn = false;

export default function ProtectedLayout() {

  if (!isLoggedIn) {
    return <Redirect href="/login" />;
  }

  return (
    <Stack>
      <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
      <Stack.Screen name="modal" options={{ presentation: "modal" }} />
      <Stack.Screen
        name="modal-with-stack"
        options={{ headerShown: false, presentation: "modal" }}
      />
    </Stack>
  );
}
```

In example if `isLoggedIn` value is false so its redirect to login if the value is true then load home screen

#### Storing the auth state

Login and logout using react [`createContext`](https://react.dev/reference/react/createContext)

Create a `authContext.tsx` File And here is the structure of file.
- In this we can add two function when we call `isLoggedIn` value change.

```tsx
import { useRouter } from "expo-router";
import { createContext, PropsWithChildren, useState } from "react";

type AuthState = {
  isLoggedIn: boolean;
  logIn: () => void;
  logOut: () => void;
};

export const AuthContext = createContext<AuthState>({
  isLoggedIn: false,
  logIn: () => {},
  logOut: () => {},
});

export function AuthProvider({ children }: PropsWithChildren) {
  const [isLoggedIn, setIsLoggedIn] = useState(false);
  const router = useRouter();

  const logIn = () => {
    setIsLoggedIn(true);
    router.replace("/");
  };

  const logOut = () => {
    setIsLoggedIn(false);
    router.replace("/login");
  };

  return (
    <AuthContext.Provider value={{ isLoggedIn, logIn, logOut }}>
      {children}
    </AuthContext.Provider>
  );
}
```

Replace you **Root layout** `React.Fragment` to `AuthContext`

#### Persist auth state across app launcher

This upside method not store in memory when we reload the app its value become default which you passed. We save it to device storage to do this we can use [async-storage](https://docs.expo.dev/versions/latest/sdk/async-storage/) in expo.
- Install async storage through docs with `npm`.
- Add and Use `AsyncStorage` in `authContect.tsx`.
- Add new `useState` **Ready**. `authStorageKey` and create `getAuthFromStorage`

```tsx
import AsyncStorage from "@react-native-async-storage/async-storage";
import { useRouter } from "expo-router";
import { createContext, PropsWithChildren, useEffect, useState } from "react";


type AuthState = {
  isLoggedIn: boolean;
  isReady: boolean;
  logIn: () => void;
  logOut: () => void;
};

const authStorageKey = "auth-key";

export const AuthContext = createContext<AuthState>({
  isLoggedIn: false,
  isReady: false,
  logIn: () => {},
  logOut: () => {},
});

export function AuthProvider({ children }: PropsWithChildren) {
  const [isReady, setIsReady] = useState(false);
  const [isLoggedIn, setIsLoggedIn] = useState(false);
  const router = useRouter();

  const storeAuthState = async (newState: { isLoggedIn: boolean }) => {
    try {
      const jsonValue = JSON.stringify(newState);
      await AsyncStorage.setItem(authStorageKey, jsonValue);
    } catch (error) {
      console.log("Error saving", error);
    }
  };

  const logIn = () => {
    setIsLoggedIn(true);
    storeAuthState({ isLoggedIn: true });
    router.replace("/");
  };

  const logOut = () => {
    setIsLoggedIn(false);
    storeAuthState({ isLoggedIn: false });
    router.replace("/login");
  };

  useEffect(() => {
    const getAuthFromStorage = async () => {
      try {
        const value = await AsyncStorage.getItem(authStorageKey);
        if (value !== null) {
          const auth = JSON.parse(value);
          setIsLoggedIn(auth.isLoggedIn);
        }
      } catch (error) {
        console.log("Error fetching from storage", error);
      }
      setIsReady(true);
    };
    getAuthFromStorage();
  }, []);

  return (
    <AuthContext.Provider value={{ isReady, isLoggedIn, logIn, logOut }}>
      {children}
    </AuthContext.Provider>
  );
}
```

- Add `isReady` condition in **(protected)** root `_layout.tsx`. 

```tsx

      <Stack.Screen
        name="modal-with-stack"
        options={{ headerShown: false, presentation: "modal" }}
      />
    </Stack>
  );
}
```

#### Show splash screen while auth is undecided

- Add Some more code in `authContext.tsx`
- Add promise and new `useEffect`
- add `SplashScreen` it is build in  expo-router this control you splash screen logo time. 

```tsx
import { SplashScreen, useRouter } from "expo-router";

SplashScreen.preventAutoHideAsync();

 useEffect(() => {
    const getAuthFromStorage = async () => {
      await new Promise((res) => setTimeout(() => res(null), 1000));

      try {
        const value = await AsyncStorage.getItem(authStorageKey);
        if (value !== null) {
          const auth = JSON.parse(value);
          setIsLoggedIn(auth.isLoggedIn);
        }
      } catch (error) {
        console.log("Error fetching from storage", error);
      }
      setIsReady(true);
    };
    getAuthFromStorage();
  }, []);
  
 useEffect(() => {
    if (isReady) {
      SplashScreen.hideAsync();
    }
  }, [isReady]);
```
