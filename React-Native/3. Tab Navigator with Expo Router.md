#### Create a bottom tabs layout

Create four pages with different names of tabs.
- `_layout.tsx` File Structure.

```tsx
import { Tabs } from "expo-router";
import React from "react";
import { StatusBar } from "expo-status-bar";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Tabs>
        <Tabs.Screen name="index" />
        <Tabs.Screen name="second" />
        <Tabs.Screen name="third" />
        <Tabs.Screen name="fourth" />
      </Tabs>
    </React.Fragment>
  );
}
```

#### Tab bar icons

We can change tabs icons using `tabBarIcon` inner options of `Tabs.Screen`.

- How to install expo icons in our project [Expo Icons](https://icons.expo.fyi/Index)

```bash
npx expo install @expo/vector-icons
```

- `_layout.tsx` File Structure.

```tsx
import { Tabs } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";
import MaterialCommunityIcons from "@expo/vector-icons/MaterialCommunityIcons";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Tabs screenOptions={{ tabBarActiveTintColor: "teal" }}>
        <Tabs.Screen
          name="index"
          options={{
```

#### Labels and title

- `_layout.tsx` File Structure.
    - In options `title` you can change top nav title.
    - In options `tabBarLabel` you can change bottom tabs name.
    - If you want to hide one or more tabs into tab screen just pass `null` in `href` tag in options. Like the last one.
    - There are some more options explore it.

```tsx
import { Tabs } from "expo-router";
import React from "react";
import { StatusBar } from "expo-status-bar";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Tabs>
        <Tabs.Screen name="index" options={{ 
        title: "Home", tabBarLabel: "Index" 
        }} />
        <Tabs.Screen name="second" options={{ title: "Second" }} />
        <Tabs.Screen name="third" options={{ title: "Third" }}/>
        <Tabs.Screen name="fourth" options={{ title: "Fouth" href: null }} />
      </Tabs>
    </React.Fragment>
  );
}
```

#### Use a stack navigator in a tab

- /index
- /second
    - /second/index
    - /second/nested
    - /second/also-nested
- /third
- /fourth

- Create a folder name `second` and add three files to control the three file create a `_layout.tsx` file inner folder.
- `/app/second/_layout.tsx` File Structure

```tsx
import { Stack } from "expo-router";
export default function SecondLayout() {
  return <Stack />;
}
```

- `/app/second/index.tsx` File Structure.

```tsx
import { View, Text } from "react-native";
import { AppText } from "@/components/AppText";
import { Link } from "expo-router";
import { Button } from "@/components/Button";

export default function SecondScreen() {
  return (
    <View className="flex-1 justify-center p-4">
      <AppText center>Second Screen</AppText>
      <Link href="/second/nested" push asChild>
        <Button title="Push to /second/nested" />
      </Link>
    </View>
  );
}
```

#### Reset tab on blur
- If we navigate to the one tab inner nested screen and click the other tab and again change the save nested tab it show same nested path. how to change on 1st page of that tab.
- Root `_layout.tsx` File Structure.
- `popToTopOnBlur` this option true when ever you check on again nested tab it automatically open first page of nested

```tsx
import { Tabs } from "expo-router";
import React from "react";
import { StatusBar } from "expo-status-bar";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Tabs>
        <Tabs.Screen name="(home)" options={{ 
        title: "Home",
        popToTopOnBlur: true,
        }} />
        <Tabs.Screen name="second" options={{ title: "Second" }} />
        <Tabs.Screen name="third" options={{ title: "Third" }}/>
        <Tabs.Screen name="fourth" options={{ title: "Fouth" }} />
      </Tabs>
    </React.Fragment>
  );
}
```

- How to change slide animation based on path.
- `app/second/_layout.tsx` File Structure.

```tsx
import { Stack, usePathname } from "expo-router";
export default function SecondLayout() {
  const pathname = usePathname();

  return (
```

#### Tab bar back action

How to go back in router
- If you click on button it go to home tab automatically.
- If you want to go back in tabs order just pass `backBehavoiur` to order in `Tabs` tag in root layout.

```tsx
import { View, Text } from "react-native";
import { useRouter } from "expo-router";
import { Button } from "@/components/Button";

export default function ThirdScreen() {
  const router = useRouter();
  return (
    <View>
      <Text>ThirdScreen</Text>
      <Button title="Back" theme="secondary" onPress={() => router.back()} />
    </View>
  );
}
```
