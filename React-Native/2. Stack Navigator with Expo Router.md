#### Passing params between screens

###### Use params with Link `useLocalSearchParams()`
- `index.tsx` file.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { Link } from "expo-router";
import { Button } from "@/components/Button";

export default function IndexScreen() {
  return (
    <View className="justify-center flex-1 p-4">
      <AppText center size="heading" bold>
        Index Screen
      </AppText>
      <Link
        href={{ pathname: "/second", params: { name: "Kadi" } }}
        push
        asChild
      >
        <Button title="Greet Kadi on  /second" theme="secondary" />
      </Link>
    </View>
  );
}
```

- `second.tsx` file.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { Link, useLocalSearchParams } from "expo-router";
import { Button } from "@/components/Button";

export default function SecondScreen() {
  const params = useLocalSearchParams<{ name?: string }>();
  
  return (
    <View className="justify-center flex-1 p-4 bg-blue-200">
      <AppText center size="heading" bold>
        Second Screen
      </AppText>
      {params.name ? (
        <AppText center>
          Hello,<AppText bold>{params.name}</AppText>
        </AppText>
      ) : null}
    </View>
  );
}
```

###### Use params with `useRouter()` and `useLocalSearchParams()`.
- `index.tsx` file.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { Link } from "expo-router";
import { Button } from "@/components/Button";

export default function IndexScreen() {
const router = useRouter();

  return (
    <View className="justify-center flex-1 p-4">
      <AppText center size="heading" bold>
        Index Screen
      </AppText>
       <Button
        title="Greet Mary on /second"
        theme="secondary"
        onPress={() =>
          router.push({ pathname: "/second", params: { name: "Mary" } })
        } />
    </View>
  );
}
```

#### Dynamic Routes

- `index.tsx` File. File directory `/proverbs/1`.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { Link, useRouter } from "expo-router";
import { Button } from "@/components/Button";
  
// /proverbs/1

export default function IndexScreen() {

  const router = useRouter();

  return (
    <View className="justify-center flex-1 p-4">
      <AppText center size="heading" bold>
        Index Screen
      </AppText>

      <Link href="/proverbs/1" push asChild>
        <Button theme="tertiary" title="Push to /proverbs/1" />
      </Link>
      
      <Link
        href={{ pathname: "/proverbs/[id]", params: { id: "2" } }}
        push
        asChild
      >
        <Button theme="tertiary" title="Push to /proverbs/2" />
      </Link>
    </View>
  );
}
```

- `[id].tsx` File. Inner `app/proverbs/[id].tsx`.

```tsx
import { View, Text } from "react-native";
import { AppText } from "@/components/AppText";
import { useLocalSearchParams } from "expo-router";

// use Data inner proverbs
const proverbs = [];
  
export default function ProverbsScreen() {
  const params = useLocalSearchParams<{ id: string }>();

  const proverb = proverbs.find((p) => p.id === params.id);

  if (!proverb) {
    return (
      <View className="flex-1 items-center justify-center">
        <AppText size="heading">Not Found</AppText>
      </View>
    );
  }

  return (
    <View className="flex-1 items-center justify-center">
      <AppText size="large" bold center>
        {proverb.proverb}
      </AppText>
      <AppText>- {proverb.source}</AppText>
    </View>
  );
}
```

#### Nested Dynamic Routes

- `index.tsx` File. File directory `/proverbs/1`.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { Link, useRouter } from "expo-router";
import { Button } from "@/components/Button";
  
// /proverbs/[category]/[productId]

export default function IndexScreen() {

  const router = useRouter();

  return (
    <View className="justify-center flex-1 p-4">
      <AppText center size="heading" bold>
        Index Screen
      </AppText>
      
      <Link href="/products/shoes/1234" push asChild>
        <Button theme="tertiary" title="Push to /products/shoes/1234" />
      </Link>
    </View>
  );
}
```

- `[productId].tsx` File. Inner `app/products/[category]/[productId].tsx`.

```tsx
import { View } from "react-native";
import { AppText } from "@/components/AppText";
import { useLocalSearchParams } from "expo-router";

export default function ProductScreen() {
  const params = useLocalSearchParams();

  return (
    <View className="flex-1 items-center justify-center">
      <AppText>{JSON.stringify(params, null, " ")}</AppText>
    </View>
  );
}
```

#### Screen Options

- `_layout.tsx` File.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";

export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Screen name="proverbs/[id]" options={{ title: "Proverb" }} />
      </Stack>
    </React.Fragment>
  );
}
```

#### Configure screen animation

- `_layout.tsx` File.
- You can add animation in `Stack.Screen` options.

```tsx
import { Stack } from "expo-router";
import "../../global.css";
import React from "react";
import { StatusBar } from "expo-status-bar";
  
export default function RootLayout() {
  return (
    <React.Fragment>
      <StatusBar style="auto" />
      <Stack>
        <Stack.Screen name="proverbs/[id]" options={{ animation: "none" }} />
      </Stack>
    </React.Fragment>
  );
}
```
