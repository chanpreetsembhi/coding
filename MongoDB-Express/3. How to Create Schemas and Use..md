1. Create a `Posts.js` File Inner models Folder.

```js
import mongoose from "mongoose";

const PostSchema = new mongoose.Schema({
  title: { type: String, required: true },
  Content: String,
});

export const Post = mongoose.model("Post", TodoSchema);
```
==You can add more values in Schema. with types like typescript.==

2. `server.js` File Structure.

```js
import mongoose from "mongoose";

import express from "express";

import { Post } from "./models/Post.js";

let conn = await mongoose.connect("mongodb://localhost:27017/todo");

const app = express();

const port = 8000;

/* Show all Posts In Get Method. */
app.get("/api/posts", async (req, res) => {
  try {
    const post = await Post.find();
    res.json(post);
  } catch (error) {
    console.error(err);
    res.status(500).json({ message: "Server Error" });
  }
});

app.listen(port, () => {
  console.log(`Example app listening on port ${port}`);
});
```
