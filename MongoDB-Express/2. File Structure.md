1. `server.js` File Structure.
   - Add your MongoDB server host **mongodb://localhost:27017** + Database name **practice**. 

```js
import mongoose from "mongoose";

let conn = await mongoose.connect("mongodb://localhost:27017/practice");
```

2. Use With Express JS.

```js
import mongoose from "mongoose";

import express from "express";

let conn = await mongoose.connect("mongodb://localhost:27017/practice");

const app = express();

const port = 8000;

app.get('/', (req, res) => {
    res.send('<h1>Hello world</h1>');
})

app.listen(port, () => {
  console.log(`Example app listening on port ${port}`);
});
```
