#### Create a Filter by Author function

```php
<?php
    $books = [
        [
            "name" => "Do Andriods Dream of Electric Sheep",
            "author" => "Philip K. Dick",
            "releaseYear" => "2010",
            "purchaseUrl" => "https//example.com"
        ],
        [
            "name" => "The Langoliers",
            "author" => "Andy Weid",
            "releaseYear" => "2006",
            "purchaseUrl" => "https//example.com"
        ],
        [
            "name" => "The Art",
            "author" => "Andy Weid",
            "releaseYear" => "2016",
            "purchaseUrl" => "https//example.com"
        ]
    ];
    
   $filterByAuthor = function ($books, $author) {
        $filteredBooks = [];
        
        foreach ($books as $book) {
             if ($book['author'] === $author) {
               $filteredBooks[] = $book;
              }
        }
      
        return $filteredBooks;
    };
    
    $filteredBooks = filterByAuthor($books, 'Andy Weid')
 ?>
 
<ul>
        <?php foreach ($filteredBooks as $book): ?>
            <li>
                <a href="<?= $book['purchaseUrl'] ?>">
                    <?= $book['name'] ?> (<?= $book['releaseYear'] ?>)                       - By <?= $book['author'] ?>
                </a>
            </li>
        <?php endforeach; ?>
    </ul>
```

#### Create a Filter by Author function with based on array key value 

```php
<?php
    $books = [
        [

            "releaseYear" => 2010,

        ],
        [
            "name" => "The Langoliers",
            "author" => "Andy Weid",
            "releaseYear" => 2006,
            "purchaseUrl" => "https//example.com"
        ],
        [
            "name" => "The Art",

            "releaseYear" => 2016,
            "purchaseUrl" => "https//example.com"
        ]
    ];
    
    function filter($items, $key, $value) {
        $filteredBooks = [];
        
        foreach ($items as $item) {
             if ($item[$key] === $value) {
               $filteredBooks[] = $item;
              }
        }
      
        return $filteredBooks;
    };
    
    $filteredBooks = filter($books, 'author', 'Andy Weid')
 ?>
 
<ul>

                <a href="<?= $book['purchaseUrl'] ?>">

                </a>
            </li>
        <?php endforeach; ?>
</ul>
```

#### Anonymous function

- show all authors books after the particular year

```php
<?php
// $fn is the short form of $function
function filter($items, $fn) {
        $filteredBooks = [];
        
        foreach ($items as $item) {
             if ($fn($item)) {
               $filteredBooks[] = $item;
              }
        }
      
        return $filteredBooks;
    };
    
    $filteredBooks = filter($books, function(book){
         return $book['releaseYear'] > 2008;
    })
 ?>
```

###### Anonymous function with PHP build in array function

```php
<?php
$filteredBooks = array_filter($books, function(book){
         return $book['releaseYear'] > 2008;
    })
?>
```
