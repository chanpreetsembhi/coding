##### How to save input value in Appwrite database.
- We add `body` value name in database also.
- We `@submit.prevent` because sometime if we don't use they send input value in website Url so always use it.

#### Vue File Structure

```vue
<script setup>
import { ref } from 'vue';
import appwriteService from "@/appwrite/config"

const body = ref("");

const handleSubmit = async () => {
  try {
    // Call the method from your Appwrite service to add data to the database
    await appwriteService.addId({ body: body.value });
    // Clear the input field after successful submission
    body.value = "";
    console.log("Data added successfully!");
  } catch (error) {
    console.error("Error adding data:", error);
  }
};
</script>

<template>
  <form @submit.prevent="handleSubmit" class="flex flex-col space-y-3 w-full">
    <input type="text" placeholder="Enter Your account Id name" v-model="body"
      class="w-full p-2 border border-gray-400 rounded-md" />
    <button type="submit" class="bg-pink-600 text-white rounded-md 
    p-2.5">Add</button>
  </form>
</template>
```

#### `config.js` file function

```js
import { Client, Databases, ID } from 'appwrite';
import conf from '@/conf/conf';

export class Service{
    client = new Client
    databases

    constructor() {
        this.client
            .setEndpoint(conf.appwriteUrl)
            .setProject(conf.appwriteProjectId)
        this.databases = new Databases(this.client)
    }
   async add({ body }) {
        try {
            return await this.databases.createDocument(
                conf.appwriteDatabaseId,
                conf.appwriteCollectionId,
                ID.unique(),
                { body }
            )
        } catch (error) {
            console.log("Error in addId:", error);
        }
    }
}

const service = new Service()
export default service
```
==Note: Always pass any object data in add function createDocument==
