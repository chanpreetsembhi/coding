#### How to delete added value

```js
import { Client, Databases, ID } from 'appwrite';
import conf from '@/conf/conf';

export class Service{
    client = new Client
    databases

    constructor() {
        this.client
            .setEndpoint(conf.appwriteUrl)
            .setProject(conf.appwriteProjectId)
        this.databases = new Databases(this.client)

    }
     async deleteId(id) {
        try {
            return await this.databases.deleteDocument(
                conf.appwriteDatabaseId,
                conf.appwriteCollectionId,
                id
            )
        } catch (error) {
            console.log("Error in deleteId:", error);
        }
    }
}

const service = new Service
export default service
```

#### Vue File Structure

```vue
<script setup>
import { onMounted, ref } from "vue";
import appwriteService from "@/appwrite/config"

const notes = ref([])

onMounted(async () => {
  const response = await appwriteService.getList()
  notes.value = response.documents
})

const deleteId = async (id) => {
  try {
    await appwriteService.deleteId(id);
    console.log("Data Seleted successfully!");
    // window.location.reload()
    notes.value = notes.value.filter((item) => item.$id !== id)
  } catch (error) {
    console.error("Error seleting data:", error);
  }
}
</script>

<template>
<div class="flex flex-col space-y-3">
    <div v-for="note in notes" :key="note.$id"
      class="bg-gray-50 border border-gray-300 rounded-lg shadow-sm flex items-center justify-between gap-3 p-4">
      {{ note.body }}
      <DeleteIcon @click="deleteId(note.$id)" class="text-red-600 size-6 ml-auto cursor-pointer" />
    </div>
  </div>
</template>
```
