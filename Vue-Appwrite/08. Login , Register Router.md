How to auth if user is not login and try to access main page in upper Url than this code redirect to login again.

#### JavaScript File Structure

```js
import { Client, Account, ID } from 'appwrite';
import conf from '@/conf/conf';

export class Service{
    client = new Client
    account

    constructor() {
        this.client
            .setEndpoint(conf.appwriteUrl)
            .setProject(conf.appwriteProjectId)
        this.account = new Account(this.client)
    }
    
     async get() {
        return await this.account.getSession("current")
     }
   }
   
const service = new Service()
export default service
```

#### Vue File Structure

`index.js` or `index.ts` Vue route file after the routes.
```js
const accountRoutes = ["login", "register"]

  

router.beforeEach(async (to, from, next) => {

  const name = to.name

  try {
    const session = await appwriteService.get()
    to.meta.session = session

    if (accountRoutes.includes(name)) {
      return next({ name: "room" })
    }
    
  } catch {
    if (!accountRoutes.includes(name)) {
      return next({ name: "login" })
    }
  }
  
 next()
 
})
```
